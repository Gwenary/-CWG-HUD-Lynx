    function playDefaultSpawnSound(mobType) {
        if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.connect(gain); gain.connect(audioCtx.destination);

        if (mobType === 'mouse') {
            // 🐭 Мышь — быстрый писк, высокая частота
            osc.type = 'sine';
            osc.frequency.setValueAtTime(900, audioCtx.currentTime);
            osc.frequency.linearRampToValueAtTime(1400, audioCtx.currentTime + 0.15);
        } else if (mobType === 'butterfly') {
            // 🦋 Бабочка — мягкий порхающий, нежный
            osc.type = 'sine';
            osc.frequency.setValueAtTime(400, audioCtx.currentTime);
            osc.frequency.linearRampToValueAtTime(700, audioCtx.currentTime + 0.3);
        } else if (mobType === 'fox') {
            // 🦊 Лиса — низкий резкий, хищный
            osc.type = 'sawtooth';
            osc.frequency.setValueAtTime(200, audioCtx.currentTime);
            osc.frequency.linearRampToValueAtTime(400, audioCtx.currentTime + 0.3);
        } else {
            // Дефолт
            osc.type = 'sine';
            osc.frequency.setValueAtTime(600, audioCtx.currentTime);
            osc.frequency.linearRampToValueAtTime(900, audioCtx.currentTime + 0.15);
        }

        gain.gain.setValueAtTime(0.2, audioCtx.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.3);
        osc.start(audioCtx.currentTime);
        osc.stop(audioCtx.currentTime + 0.3);
    }

    }

    const FIGHT_THRESHOLDS = [4, 16, 58, 112, 208, 358, 604, 1000];
    const MED_THRESHOLDS   = [30, 90, 170, 270, 390, 520, 660, 810];

    function getSkillInfo(exp, thresholds) {
        let level = 1;
        let prevThreshold = 0;
        for (let i = 0; i < thresholds.length; i++) {
            if (exp < thresholds[i]) {
                level = i + 1;
                const curExp = Math.floor(exp - prevThreshold);
                const maxExp = thresholds[i] - prevThreshold;
                return { level, curExp, maxExp };
            }
            prevThreshold = thresholds[i];
        }
        return { level: thresholds.length + 1, curExp: Math.floor(exp - prevThreshold), maxExp: null };
    }

    // === Обработка WS ===
    function handlePayload(payload) {
        const now = Date.now();
        const entities = payload.entities;
        if (!entities || !Array.isArray(entities)) return;

        entities.forEach(entity => {
            if (!entity || !entity.type) return;

            if (MOB_TYPES[entity.type]) {
                const key = `${entity.type}_${entity.position?.[0]}_${entity.position?.[1]}`;
                const last = recentAlerts.get(key) || 0;
                if (now - last > 5000) {
                    recentAlerts.set(key, now);
                    playSpawnSound(entity.type);
                    showSpawnBanner(MOB_TYPES[entity.type]);
                }
            }
        });
    }

    function handleNewEntities(data) {
        let entities = data.payload || data.entities || data.data || (Array.isArray(data) ? data : null);
        if (!entities || !Array.isArray(entities)) return;
        const now = Date.now();
        entities.forEach(entity => {
            const type = entity.type;
            if (!type || !MOB_TYPES[type]) return;
            const key = entity.id || `${type}_${entity.position?.[0]}_${entity.position?.[1]}`;
            const last = recentAlerts.get(key) || 0;
            if (now - last > 5000) {
                recentAlerts.set(key, now);
                playSpawnSound(type);
                showSpawnBanner(MOB_TYPES[type]);
            }
        });
    }

    let _cachedCanvas = null;
    let _cachedRect = null;
    let _rectLastUpdate = 0;

    function getCachedCanvasRect() {
        if (!_cachedCanvas) _cachedCanvas = document.querySelector('#root > canvas');
        if (!_cachedCanvas) return null;
        const now = Date.now();
        if (now - _rectLastUpdate > 500) {
            _cachedRect = _cachedCanvas.getBoundingClientRect();
            _rectLastUpdate = now;
        }
        return _cachedRect;
    }

    function getSpritePosition(sprite) {
        const rect = getCachedCanvasRect();
        if (!rect || !sprite) return null;
        const scaleX = rect.width / (_cachedCanvas.clientWidth || _cachedCanvas.width);
        const scaleY = rect.height / (_cachedCanvas.clientHeight || _cachedCanvas.height);
        let gp = { x: 0, y: 0 };
        if (typeof sprite.getGlobalPosition === 'function') {
            gp = sprite.getGlobalPosition();
        } else if (sprite.worldTransform) {
            gp.x = sprite.worldTransform.tx;
            gp.y = sprite.worldTransform.ty;
        }
        return {
            x: rect.left + gp.x * scaleX,
            y: rect.top + gp.y * scaleY
        };
    }

    function getSpriteCenterFromBounds(sprite) {
        const rect = getCachedCanvasRect();
        if (!rect || !sprite) return null;
        const scaleX = rect.width / (_cachedCanvas.clientWidth || _cachedCanvas.width);
        const scaleY = rect.height / (_cachedCanvas.clientHeight || _cachedCanvas.height);
        const bounds = sprite.getBounds?.();
        if (!bounds) return null;
        return {
            x: rect.left + ((bounds.minX + bounds.maxX) / 2) * scaleX,
            y: rect.top + ((bounds.minY + bounds.maxY) / 2) * scaleY,
        };
    }

        const usedUIDs = new Set();

        for (const sprite of root.children) {
            const uid = sprite.uid;
            if (!uid) continue;

            const isCat = getShowCatNames() && catUidToDataCache.has(uid);
            const isMob = !isCat && MOB_LABELS.has(sprite.label);

            if (!isMob && !isCat) continue;

            usedUIDs.add(uid);

            if (isMob && getHighlightMobs()) {
                let overlay = activeOverlays.get(uid);
                if (!overlay || !overlay.classList.contains('cw-mob-glow')) {
                    if (overlay) overlay.remove();
                    overlay = document.createElement('div');
                    overlay.className = 'cw-mob-glow';
                    overlay.style.cssText = `
                        position: fixed;
                        pointer-events: none;
                        z-index: 100;
                        width: 60px;
                        height: 60px;
                        background: radial-gradient(ellipse at center, transparent 20%, rgba(255,220,80,0.7) 50%, rgba(255,120,0,0.4) 70%, transparent 100%); 
                        border-radius: 50%;
                        transform: translate(-50%, -50%);
                        filter: blur(3px);
                        mix-blend-mode: screen;
                    `;
                    document.body.appendChild(overlay);
                    activeOverlays.set(uid, overlay);
                }
                const mobType = Object.entries(MOB_TYPES).find(([,v]) => v.includes(sprite.label?.slice(2)))?.[0];
                const mobColor = (mobType && CFG.mobColors[mobType]) || '#ffdc50';
                const r = parseInt(mobColor.slice(1,3),16);
                const g = parseInt(mobColor.slice(3,5),16);
                const b = parseInt(mobColor.slice(5,7),16);
                overlay.style.background = `radial-gradient(ellipse at center, transparent 20%, rgba(${r},${g},${b},0.7) 50%, rgba(${Math.round(r*0.5)},${Math.round(g*0.5)},${Math.round(b*0.5)},0.4) 70%, transparent 100%)`;

                const center = getSpriteCenterFromBounds(sprite);
                if (center) {
                    overlay.style.left = center.x + 'px';
                    overlay.style.top = center.y + 'px';
                }

            } else if (isCat) {
                // Скрываем имена когда открыт чат
                if (chatOpen) {
                    const existing = activeOverlays.get(uid);
                    if (existing) existing.style.visibility = 'hidden';
                    usedUIDs.add(uid);
                    continue;
                }
                const playerData = catUidToDataCache.get(uid);
                let overlay = activeOverlays.get(uid);

                if (!overlay || !overlay.classList.contains('cw-cat-name')) {
                    if (overlay) overlay.remove();
                    overlay = document.createElement('div');
                    overlay.className = 'cw-cat-name';
                    overlay.innerHTML = playerData.clanName
                        ? `${playerData.name} <span class="clan">[${playerData.clanName}]</span>`
                        : playerData.name;
                    overlay.style.cssText = `
                        position: fixed;
                        pointer-events: none;
                        z-index: 50;
                        background: rgba(20, 20, 30, 0.75);
                        backdrop-filter: blur(8px);
                        border-radius: 12px;
                        padding: 4px 10px;
                        color: white;
                        font-weight: bold;
                        font-size: 13px;
                        text-shadow: 0 2px 4px black;
                        border: 1px solid rgba(255,255,255,0.2);
                        transform: translate(-50%, -80%);
                        white-space: nowrap;
                    `;
                    document.body.appendChild(overlay);
                    activeOverlays.set(uid, overlay);
                }

                const pos = getSpritePosition(sprite);
                if (pos) {
                    overlay.style.visibility = 'visible';
                    overlay.style.left = pos.x + 'px';
                    overlay.style.top = (pos.y - 55) + 'px';
                }
            }
        }

        for (const [uid, overlay] of activeOverlays) {
            if (!usedUIDs.has(uid)) {
                overlay.remove();
                activeOverlays.delete(uid);
            }
        }
    }

    function getSticker(hour) {
        if (hour >= 6 && hour < 12) return '🌅';
        if (hour >= 12 && hour < 18) return '☀️';
        if (hour >= 18 && hour < 24) return '🌆';
        return '🌙';
    }

    function getGameTimeObject() {
        const date = window.CWGPlayground?.getDate?.() || window.CWGPlayground?.getVirtualTime?.();
        if (date && Array.isArray(date) && date.length >= 5) return { hour: Number(date[3]), min: Number(date[4]), valid: true };
        if (date instanceof Date && !isNaN(date)) return { hour: date.getHours(), min: date.getMinutes(), valid: true };
        if (typeof date === 'string' && /^\d{1,2}:\d{2}$/.test(date.trim())) {
            const [h,m] = date.trim().split(':').map(Number);
            return { hour: h, min: m, valid: true };
        }
        if (typeof date === 'object' && date.hour !== undefined && date.minute !== undefined) {
            return { hour: Number(date.hour), min: Number(date.minute), valid: true };
        }
        return { hour: 0, min: 0, valid: false };
    }

    function updateHUD() {
        if (timeElement && stickerElement) {
            const t = getGameTimeObject();
            if (t.valid) {
                timeElement.textContent = `${String(t.hour).padStart(2,'0')}:${String(t.min).padStart(2,'0')}`;
                stickerElement.textContent = getSticker(t.hour);
            } else {
                timeElement.textContent = '??:??';
                stickerElement.textContent = '❓';
            }
        }
        if (hpFill && hpLabel) {
            const cat = window.CWGPlayground?.me?.();
            if (cat) {
                let hp = cat.hp?.value ?? cat.hp;
                hp = Number(hp);
                let maxHp = cat.maxHp || cat.hp?.max || 2000;
                maxHp = Number(maxHp);
                if (!isNaN(hp) && maxHp > 0) {
                    const pct = (hp / maxHp) * 100;
                    hpFill.style.width = `${Math.min(100, Math.max(0, pct))}%`;
                    hpFill.style.background = pct > 65
                        ? 'linear-gradient(90deg,#43a047,#66bb6a)'
                        : pct > 40
                            ? 'linear-gradient(90deg,#ef6c00,#ffa726)'
                            : 'linear-gradient(90deg,#c62828,#ef5350)';
                    hpLabel.textContent = `${Math.round(hp)}/${Math.round(maxHp)}`;
                    if (lastHp !== null) {
                        const delta = lastHp - hp;
                        if (delta > 10) {
                            playHitSound();
                            showHitBanner(delta);
                            cdStartTime = Date.now(); // сбрасываем КД при ударе
                        }
                    }
                    lastHp = hp;
                } else {
                    hpLabel.textContent = '?/?';
                    lastHp = null;
                }
            } else {
                hpLabel.textContent = '?/?';
                lastHp = null;
            }
        }
        // ========== КД удара ==========
        if (cdFill && cdLabel) {
            if (cdStartTime === null) {
                // Ещё не было удара — полная шкала
                cdFill.style.width = '100%';
                cdFill.style.background = 'linear-gradient(90deg,#5c6bc0,#42a5f5)';
                cdLabel.textContent = 'Готово';
            } else {
                const elapsed = Date.now() - cdStartTime;
                const pct = Math.min(1, elapsed / CD_DURATION);
                cdFill.style.width = `${pct * 100}%`;
                if (pct >= 1) {
                    cdFill.style.background = 'linear-gradient(90deg,#5c6bc0,#42a5f5)';
                    cdLabel.textContent = 'Готово';
                } else {
                    cdFill.style.background = 'linear-gradient(90deg,#b71c1c,#ef5350)';
                    cdLabel.textContent = `${((CD_DURATION - elapsed) / 1000).toFixed(1)}s`;
                }
            }
        }
    }

            const tog = document.createElement('label');

        if (skillsPanel) skillsPanel.style.display = CFG.showSkillsPanel ? 'flex' : 'none';

        if (timeElement) timeElement.parentElement.style.display = CFG.showClock ? 'flex' : 'none';
    }
